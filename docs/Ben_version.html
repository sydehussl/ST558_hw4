Build a function to do this for us

news_search <- function(search, start_date, api_key) {
  # get url
  response <- GET(url = "https://newsapi.org/v2/everything",
                  query = list(q = search,
                               from = start_date,
                               language = "en",
                               pageSize = 50,
                               apiKey = api_key))
  
  # find article information
  parsed <- fromJSON(rawToChar(response$content))
  
  # make it a tibble
  articles <- parsed$articles |>
    as_tibble()

  return(articles)
}
